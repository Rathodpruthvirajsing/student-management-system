--- users ---

Field: name | Type: varchar(100) | Null: NO | Default: NULL
Field: email | Type: varchar(150) | Null: NO | Default: NULL
Field: password | Type: varchar(255) | Null: NO | Default: NULL
Field: role | Type: enum('admin','teacher','student') | Null: YES | Default: admin
Field: created_at | Type: timestamp | Null: NO | Default: current_timestamp()
